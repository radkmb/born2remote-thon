                #include "ft_stock_str.h"


                    void	ft_putnbr(int nb)
                    {
                        char minus_char;
                        char temp_nb;

                        minus_char = '-';
                        if (nb == -2147483648)
                            write(1, "-2147483648", 11);
                        else
                        {
                            if (nb < 0)
                            {
                                nb = nb * (-1);
                                write(1, &minus_char, 1);
                            }
                            if (nb < 10)
                            {
                                temp_nb = nb + '0';
                                write(1, &temp_nb, 1);
                            }
                            else
                            {
                                ft_putnbr(nb / 10);
                                ft_putnbr(nb % 10);
                            }
                        }
                    }

                    void	ft_putstr(char *str)
                    {
                        int i;

                        i = 0;
                        while (str[i])
                        {
                            write(1, &str[i], 1);
                            i++;
                        }
                    }

                    void	ft_show_tab(struct s_stock_str *par)
                    {
                        int i;

                        i = 0;
                        while (par[i].str != 0)
                        {
                            ft_putstr(par[i].str);
                            write(1, "\n", 1);
                            ft_putnbr(par[i].size);
                            write(1, "\n", 1);
                            ft_putstr(par[i].copy);
                            write(1, "\n", 1);
                            i++;
                        }
                    }
